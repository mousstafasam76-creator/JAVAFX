@echo off
cd /d C:\inapp-1.0.0_desktop\JAVAFX
echo Compilation...
dir /s /B src\main\java\*.java > sources.txt
javac --module-path "C:\javafx-sdk-21.0.2\lib" --add-modules javafx.controls,javafx.fxml -d out @sources.txt
if errorlevel 1 (
    echo Erreur de compilation
    pause
    exit /b
)
del sources.txt
echo Copie des ressources...
xcopy src\main\resources out /E /I /Y > nul
echo Lancement...
java --module-path "C:\javafx-sdk-21.0.2\lib" --add-modules javafx.controls,javafx.fxml -cp out com.inapp.MainApplication
pause
